// Fit data buckets let tops = []; let bottoms = []; let hats = []; let accs = []; // IMPORTANT: // Add a BLANK option for hat/accessories so they can be "empty" but still swipe. // Blank is represented by empty src "". const BLANK = { name: "", src: "" }; // Current indices let idxTop = 0, idxBot = 0, idxHat = 0, idxAcc = 0; // Render functions function setImg(imgEl, slotEl, src){ const trimmed = (src || "").trim(); if (!trimmed) { imgEl.removeAttribute("src"); imgEl.style.display = "none"; slotEl.classList.add("blank"); } else { imgEl.src = trimmed; imgEl.style.display = "block"; slotEl.classList.remove("blank"); } } function render(){ setImg(document.getElementById("topImg"), document.getElementById("topSlot"), tops[idxTop]?.src); setImg(document.getElementById("botImg"), document.getElementById("botSlot"), bottoms[idxBot]?.src); setImg(document.getElementById("hatImg"), document.getElementById("hatSlot"), hats[idxHat]?.src); setImg(document.getElementById("accImg"), document.getElementById("accSlot"), accs[idxAcc]?.src); } // Swipe handler (works even when slot is blank) function attachSwipe(slotEl, getList, getIndex, setIndex){ let startX = 0; let startY = 0; slotEl.addEventListener("touchstart", (e) => { const t = e.touches[0]; startX = t.clientX; startY = t.clientY; }, { passive:true }); slotEl.addEventListener("touchend", (e) => { const t = e.changedTouches[0]; const endX = t.clientX; const endY = t.clientY; const dx = endX - startX; const dy = endY - startY; // Only treat as swipe if mostly horizontal if (Math.abs(dx) < 45 || Math.abs(dx) < Math.abs(dy)) return; const list = getList(); if (!list || list.length <= 1) return; const cur = getIndex(); if (dx < 0) { // swipe left -> next setIndex((cur + 1) % list.length); } else { // swipe right -> prev setIndex((cur - 1 + list.length) % list.length); } render(); }, { passive:true }); } // Random + Save function randInt(n){ return Math.floor(Math.random() * n); } function randomize(){ if (tops.length) idxTop = randInt(tops.length); if (bottoms.length) idxBot = randInt(bottoms.length); if (hats.length) idxHat = randInt(hats.length); if (accs.length) idxAcc = randInt(accs.length); render(); } // Save: downloads a screenshot-like image using html2canvas-style approach would need a lib. // Instead: use native share screenshot (iOS) OR we can add a lightweight save later. // For now we’ll just tell the user to take a screenshot. function saveFit(){ alert("Tip: Take a screenshot to save your fit.\n(If you want a real 'Save Image' button, tell me and I’ll add it.)"); } document.getElementById("randomBtn").addEventListener("click", randomize); document.getElementById("saveBtn").addEventListener("click", saveFit); // Load sheet and build lists fetch(sheetURL) .then(r => r.text()) .then(csv => { const rows = csv.trim().split("\n").slice(1); const items = rows .map(line => splitCSVLine(line)) .filter(cols => cols.length >= 9) .map(cols => ({ name: (cols[1] || "").trim(), category: (cols[2] || "").trim(), // IMPORTANT: this assumes your image column already contains FULL URLs or valid paths // If you're using filenames only, change src to your folder path (e.g. "images/" + cols[8]). src: (cols[8] || "").trim().split("|")[0].trim() // use first image for fit })); tops = items.filter(i => i.category === "Tops"); bottoms = items.filter(i => i.category === "Bottoms"); // Headwear + Accessories lists hats = [BLANK, ...items.filter(i => i.category === "Headwear")]; accs = [BLANK, ...items.filter(i => i.category === "Accessories")]; // Start with blank hat/acc idxHat = 0; idxAcc = 0; idxTop = 0; idxBot = 0; render(); // Attach swipes (blank still swipeable because list includes BLANK) attachSwipe(document.getElementById("topSlot"), () => tops, () => idxTop, (v) => idxTop = v); attachSwipe(document.getElementById("botSlot"), () => bottoms, () => idxBot, (v) => idxBot = v); attachSwipe(document.getElementById("hatSlot"), () => hats, () => idxHat, (v) => idxHat = v); attachSwipe(document.getElementById("accSlot"), () => accs, () => idxAcc, (v) => idxAcc = v); });
